### 매도 조건식
```python
if 등락율 > 29.7:
    매도 = True
elif 수익률 >= 7.0:
    매도 = True
elif 수익률 <= -2.2:
    매도 = True
elif 최고수익률 - 수익률 >= 5.0:
    매도 = True
elif 보유시간 > 270:
    매도 = True

elif 시가총액 < 1000:
    if 복합지표1 < 0.8:
        매도 = True
    elif 체결강도 < 75 and 초당매도수량 > 초당매수수량 * 1.8:
        매도 = True

elif 시가총액 < 2000:
    if 복합지표1 < 0.9:
        매도 = True
    elif 체결강도 < 80 and 초당매도수량 > 초당매수수량 * 1.7:
        매도 = True

elif 시가총액 < 3500:
    if 복합지표1 < 1.0:
        매도 = True
    elif 체결강도 < 85 and 초당매도수량 > 초당매수수량 * 1.6:
        매도 = True

elif 시가총액 < 5000:
    if 복합지표1 < 1.1:
        매도 = True
    elif 체결강도 < 90 and 초당매도수량 > 초당매수수량 * 1.5:
        매도 = True

else:
    if 복합지표1 < 1.2:
        매도 = True
    elif 체결강도 < 95 and 초당매도수량 > 초당매수수량 * 1.4:
        매도 = True
```
