## 소개

이 문서는 장초 09:00:00~09:20:00 전체 구간에서 이동평균선 돌파를 활용한 스캘핑 전략입니다.
단기 이동평균(60틱)을 상향 돌파하면서 거래량이 동반될 때 진입하여 빠른 수익을 실현합니다.

## 가이드라인

- 모든 조건식은 Python으로 작성되어 있으며, 주식 거래의 자동화를 목표로 합니다.
- 각 조건식은 특정 조건을 만족할 때 매수 또는 매도를 실행합니다.

### 저장 이름 규칙

1. **기본 규칙**:
   - `C`: Condition을 의미합니다.
   - `MABS`: MA Breakout Scalping을 의미합니다.
   - `B`: 매수 조건식을 의미합니다.
   - `S`: 매도 조건식을 의미합니다.

2. **예시**:
   - `C_MABS_B`: MA Breakout Scalping 매수 조건식
   - `C_MABS_S`: MA Breakout Scalping 매도 조건식

# Condition - MA Breakout Scalping

## 조건식

### 매수 조건식

```python
# 이동평균 돌파 스캘핑 전략 (09:00:00 ~ 09:20:00)

# 시간 조건
if not (90000 <= 시분초 <= 92000):
    매수 = False

# 데이터 충분성 확인 (이동평균 계산을 위해)
elif not (데이터길이 > 90):
    매수 = False

# 가격 및 등락율 조건
elif not (2000 < 현재가 <= 50000):
    매수 = False
elif not (1.0 < 등락율 <= 18.0):
    매수 = False

# VI 및 라운드피겨 제외
elif not (현재가 < VI아래5호가):
    매수 = False
elif 라운드피겨위5호가이내:
    매수 = False

# 이동평균 돌파 조건 (핵심)
# 현재가가 60틱 이동평균을 상향 돌파
elif not (현재가 > 이동평균(60)):
    매수 = False
elif not (현재가N(1) <= 이동평균(60, 1) or 현재가N(2) <= 이동평균(60, 2)):
    매수 = False

# 이동평균 상승 추세 확인
elif not (이동평균(60) > 이동평균(60, 1)):
    매수 = False
elif not (이동평균(60, 1) > 이동평균(60, 2)):
    매수 = False

# 단기 이평 > 장기 이평 (정배열)
elif not (이동평균(30) > 이동평균(90)):
    매수 = False

# 거래량 동반 조건
elif not (초당거래대금 > 초당거래대금평균(60) * 1.5):
    매수 = False
elif not (초당거래대금 > 초당거래대금N(1)):
    매수 = False

# 체결강도 조건
elif not (체결강도 > 130):
    매수 = False
elif not (체결강도평균(30) > 115):
    매수 = False

# 상승 각도 조건
elif not (등락율각도(30) > 8):
    매수 = False

# 시가총액별 세분화 조건
elif 시가총액 < 2500:
    # 소형주
    if not ((초당매수수량 - 초당매도수량) * 현재가 / 1_000_000 > 60):
        매수 = False
    elif not (최고초당매수수량(30) > 최고초당매도수량(30) * 2.5):
        매수 = False
    elif not (당일거래대금 > 30 * 100):
        매수 = False
    elif not (전일비 > 70):
        매수 = False
    elif not (전일동시간비 > 100):
        매수 = False
    elif not (회전율 > 2):
        매수 = False
    elif not (현재가 > 고가 - (고가 - 저가) * 20 / 100):
        매수 = False
    elif not (매수총잔량 > 매도총잔량):
        매수 = False

else:
    # 중대형주
    if not ((초당매수수량 - 초당매도수량) * 현재가 / 1_000_000 > 100):
        매수 = False
    elif not (최고초당매수수량(30) > 최고초당매도수량(30) * 2.0):
        매수 = False
    elif not (당일거래대금 > 80 * 100):
        매수 = False
    elif not (전일비 > 50):
        매수 = False
    elif not (전일동시간비 > 80):
        매수 = False
    elif not (회전율 > 1.5):
        매수 = False
    elif not (현재가 > 고가 - (고가 - 저가) * 30 / 100):
        매수 = False
    elif not (매수총잔량 > 매도총잔량 * 0.8):
        매수 = False

# 최종 매수 실행
if 매수:
    self.Buy(종목코드, 종목명, 매수수량, 현재가, 매도호가1, 매수호가1, 데이터길이)
```

### 매도 조건식

```python
# 이동평균 이탈 및 익절 전략
if 등락율 > 29.5:
    매도 = True
elif 수익률 >= 3.0:
    매도 = True
elif 수익률 <= -2.5:
    매도 = True
elif 최고수익률 - 수익률 >= 2.0:
    매도 = True
elif 보유시간 > 200:  # 3분 20초 초과 보유 시 청산
    매도 = True

# 이동평균 하향 이탈 (핵심 매도 조건)
elif 현재가 < 이동평균(60) * 0.995:
    if 수익률 > 0:
        매도 = True
elif 현재가N(1) >= 이동평균(60, 1) and 현재가 < 이동평균(60):
    if 수익률 > 0.3:
        매도 = True

# 이동평균 하락 전환
elif 이동평균(60) < 이동평균(60, 1) and 이동평균(60, 1) < 이동평균(60, 2):
    if 수익률 > 0.5:
        매도 = True

# 체결강도 약화
elif 체결강도 < 100:
    매도 = True
elif 체결강도평균(20) < 105:
    if 수익률 > 0.5:
        매도 = True

# 시가총액별 세분화 매도 조건
elif 시가총액 < 2500:
    # 소형주: 빠른 청산
    if 1.0 <= 수익률 < 3.0 and (초당매도수량 - 초당매수수량) >= 매수총잔량 * 0.7:
        매도 = True
    elif 초당거래대금 < 초당거래대금평균(30) * 0.6:
        매도 = True
    elif 등락율각도(20) < 3:
        매도 = True

else:
    # 중대형주
    if 1.0 <= 수익률 < 3.0 and (초당매도수량 - 초당매수수량) >= 매수총잔량 * 0.6:
        매도 = True
    elif 초당거래대금 < 초당거래대금평균(30) * 0.7:
        매도 = True
    elif 등락율각도(20) < 5:
        매도 = True

if 매도:
    self.Sell(종목코드, 종목명, 매도수량, 현재가, 매도호가1, 매수호가1, 강제청산)
```
