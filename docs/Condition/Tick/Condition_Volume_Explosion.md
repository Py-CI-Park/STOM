## 소개

이 문서는 장초 09:10:00~09:15:00 구간에서 거래대금의 폭발적 증가를 감지하여 매수하는 전략입니다.
평균 대비 3배 이상의 거래대금이 발생할 때 강한 모멘텀으로 판단하고 진입합니다.

## 가이드라인

- 모든 조건식은 Python으로 작성되어 있으며, 주식 거래의 자동화를 목표로 합니다.
- 각 조건식은 특정 조건을 만족할 때 매수 또는 매도를 실행합니다.

### 저장 이름 규칙

1. **기본 규칙**:
   - `C`: Condition을 의미합니다.
   - `VE`: Volume Explosion을 의미합니다.
   - `B`: 매수 조건식을 의미합니다.
   - `S`: 매도 조건식을 의미합니다.

2. **예시**:
   - `C_VE_B`: Volume Explosion 매수 조건식
   - `C_VE_S`: Volume Explosion 매도 조건식

# Condition - Volume Explosion

## 조건식

### 매수 조건식

```python
# 거래대금 폭발 전략 (09:10:00 ~ 09:15:00)

# 시간 조건
if not (91000 <= 시분초 <= 91500):
    매수 = False

# 가격 및 등락율 조건
elif not (2000 < 현재가 <= 45000):
    매수 = False
elif not (1.0 < 등락율 <= 20.0):
    매수 = False

# VI 및 라운드피겨 제외
elif not (현재가 < VI아래5호가):
    매수 = False
elif 라운드피겨위5호가이내:
    매수 = False

# 거래대금 폭발 조건 (핵심)
elif not (초당거래대금 > 초당거래대금평균(60) * 3.0):
    매수 = False
elif not (초당거래대금 > 초당거래대금N(1) * 1.2):
    매수 = False
elif not (초당거래대금 > 초당거래대금N(2) * 1.2):
    매수 = False

# 거래대금 각도 조건
elif not (당일거래대금각도(30) > 20):
    매수 = False

# 체결강도 조건
elif not (체결강도 > 120):
    매수 = False
elif not (체결강도평균(20) > 110):
    매수 = False

# 전일비 및 전일동시간비 조건
elif not (전일비 > 100):
    매수 = False
elif not (전일동시간비 > 150):
    매수 = False

# 시가총액별 세분화 조건
elif 시가총액 < 1500:
    # 초소형주: 매우 강한 폭발력 요구
    if not ((초당매수수량 - 초당매도수량) * 현재가 / 1_000_000 > 100):
        매수 = False
    elif not (등락율각도(30) > 18):
        매수 = False
    elif not (최고초당매수수량(30) > 최고초당매도수량(30) * 4):
        매수 = False
    elif not (당일거래대금 > 50 * 100):
        매수 = False
    elif not (회전율 > 3):
        매수 = False
    elif not (현재가 > 고가 - (고가 - 저가) * 10 / 100):
        매수 = False

elif 시가총액 < 3000:
    # 소형주
    if not ((초당매수수량 - 초당매도수량) * 현재가 / 1_000_000 > 120):
        매수 = False
    elif not (등락율각도(30) > 15):
        매수 = False
    elif not (최고초당매수수량(30) > 최고초당매도수량(30) * 3):
        매수 = False
    elif not (당일거래대금 > 80 * 100):
        매수 = False
    elif not (회전율 > 2.5):
        매수 = False
    elif not (현재가 > 고가 - (고가 - 저가) * 15 / 100):
        매수 = False

else:
    # 중대형주
    if not ((초당매수수량 - 초당매도수량) * 현재가 / 1_000_000 > 150):
        매수 = False
    elif not (등락율각도(30) > 12):
        매수 = False
    elif not (최고초당매수수량(30) > 최고초당매도수량(30) * 2):
        매수 = False
    elif not (당일거래대금 > 150 * 100):
        매수 = False
    elif not (회전율 > 2):
        매수 = False
    elif not (현재가 > 고가 - (고가 - 저가) * 20 / 100):
        매수 = False

# 최종 매수 실행
if 매수:
    self.Buy(종목코드, 종목명, 매수수량, 현재가, 매도호가1, 매수호가1, 데이터길이)
```

### 매도 조건식

```python
# 폭발 후 급락 방어 전략
if 등락율 > 29.5:
    매도 = True
elif 수익률 >= 3.5:
    매도 = True
elif 수익률 <= -2.0:
    매도 = True
elif 최고수익률 - 수익률 >= 2.5:
    매도 = True
elif 보유시간 > 150:  # 2분 30초 초과 보유 시 청산
    매도 = True

# 거래대금 급감 조건
elif 초당거래대금 < 초당거래대금평균(30) * 0.5:
    if 수익률 > 0:
        매도 = True

# 체결강도 급락 조건
elif 체결강도 < 90 and 체결강도평균(20) > 110:
    매도 = True

# 시가총액별 세분화 매도 조건
elif 시가총액 < 1500:
    # 초소형주: 매우 빠른 청산
    if 1.0 <= 수익률 < 3.5 and (초당매도수량 - 초당매수수량) >= 매수총잔량 * 0.9:
        매도 = True
    elif 당일거래대금각도(20) < 5:
        매도 = True
    elif 등락율각도(20) < 8:
        매도 = True

elif 시가총액 < 3000:
    # 소형주
    if 1.0 <= 수익률 < 3.5 and (초당매도수량 - 초당매수수량) >= 매수총잔량 * 0.7:
        매도 = True
    elif 당일거래대금각도(20) < 8:
        매도 = True
    elif 등락율각도(20) < 10:
        매도 = True

else:
    # 중대형주
    if 1.0 <= 수익률 < 3.5 and (초당매도수량 - 초당매수수량) >= 매수총잔량 * 0.6:
        매도 = True
    elif 당일거래대금각도(20) < 10:
        매도 = True
    elif 등락율각도(20) < 11:
        매도 = True

if 매도:
    self.Sell(종목코드, 종목명, 매도수량, 현재가, 매도호가1, 매수호가1, 강제청산)
```
