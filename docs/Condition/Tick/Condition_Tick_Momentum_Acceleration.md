# 조건식 - 등락율 모멘텀 가속 전략

## 소개
**조건식 이름**: 등락율 모멘텀 가속 전략
**거래 시간**: 90000 ~ 92000
**목표**: 등락율 가속도 증가로 상승 가속 포착

## 조건식

### 매수 조건식
```python
등락율변화 = 등락율 - 등락율N(1)
등락율가속도 = (등락율 - 등락율N(1)) - (등락율N(1) - 등락율N(2))

if not (90000 <= 시분초 < 92000):
    매수 = False
elif not (1100 < 현재가 <= 41000):
    매수 = False
elif not (2.0 < 등락율 <= 28.2):
    매수 = False
elif not (등락율가속도 > 0.15):
    매수 = False
elif not (등락율변화 > 0.08):
    매수 = False
elif 시가총액 < 1500:
    if not (등락율각도(17) > 등락율각도(17, 1) > 등락율각도(17, 2)):
        매수 = False
    elif not (현재가 > 현재가N(1) > 현재가N(2) > 현재가N(3)):
        매수 = False
    elif not (체결강도 > 164):
        매수 = False
    elif not (초당거래대금 > 초당거래대금평균(23) * 3.0):
        매수 = False
    elif not (전일비 > 48):
        매수 = False
    elif not (당일거래대금 > 18 * 100):
        매수 = False
elif 시가총액 < 3000:
    if not (등락율각도(17) > 등락율각도(17, 1) > 등락율각도(17, 2)):
        매수 = False
    elif not (현재가 > 현재가N(1) > 현재가N(2) > 현재가N(3)):
        매수 = False
    elif not (체결강도 > 154):
        매수 = False
    elif not (초당거래대금 > 초당거래대금평균(23) * 2.6):
        매수 = False
    elif not (전일비 > 58):
        매수 = False
    elif not (당일거래대금 > 44 * 100):
        매수 = False
else:
    if not (등락율각도(17) > 등락율각도(17, 1) > 등락율각도(17, 2)):
        매수 = False
    elif not (현재가 > 현재가N(1) > 현재가N(2) > 현재가N(3)):
        매수 = False
    elif not (체결강도 > 144):
        매수 = False
    elif not (초당거래대금 > 초당거래대금평균(23) * 2.3):
        매수 = False
    elif not (전일비 > 68):
        매수 = False
    elif not (당일거래대금 > 70 * 100):
        매수 = False
```

### 매도 조건식
```python
등락율감속 = (등락율 - 등락율N(1)) - (등락율N(1) - 등락율N(2))

if 등락율 > 29.6:
    매도 = True
elif 수익률 >= 6.5:
    매도 = True
elif 수익률 <= -2.5:
    매도 = True
elif 최고수익률 - 수익률 >= 4.5:
    매도 = True
elif 보유시간 > 260:
    매도 = True
elif 등락율감속 < -0.12:
    if 체결강도 < 90:
        매도 = True
elif 등락율각도(17) < 등락율각도(17, 1) < 등락율각도(17, 2):
    매도 = True
```
