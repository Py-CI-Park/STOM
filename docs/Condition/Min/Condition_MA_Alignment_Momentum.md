## 소개

이 문서는 전체 거래시간 09:00~15:18 구간에서 이동평균선 정배열을 활용한 추세 추종 분봉 전략입니다.
단기, 중기, 장기 이동평균선이 정배열되고 모두 상승 중일 때 강한 상승 추세로 판단하고 진입합니다.

## 가이드라인

- 모든 조건식은 Python으로 작성되어 있으며, 주식 거래의 자동화를 목표로 합니다.
- 각 조건식은 특정 조건을 만족할 때 매수 또는 매도를 실행합니다.
- 분봉 데이터 기반으로 900~1518 (09:00~15:18) 시간에만 거래합니다.

### 저장 이름 규칙

1. **기본 규칙**:
   - `C`: Condition을 의미합니다.
   - `MAM`: MA Alignment Momentum을 의미합니다.
   - `B`: 매수 조건식을 의미합니다.
   - `S`: 매도 조건식을 의미합니다.
   - `M`: Min (분봉) 조건식을 의미합니다.

2. **예시**:
   - `C_MAM_M_B`: MA Alignment Momentum 분봉 매수 조건식
   - `C_MAM_M_S`: MA Alignment Momentum 분봉 매도 조건식

# Condition - MA Alignment Momentum

## 조건식

### 매수 조건식

```python
# 이동평균 정배열 모멘텀 전략 (09:00 ~ 15:18)

# 시간 조건
if not (90000 <= 시분초 <= 151800):
    매수 = False

# 데이터 충분성 확인 (이동평균 계산을 위해)
elif not (데이터길이 > 80):
    매수 = False

# 가격 및 등락율 조건
elif not (2000 < 현재가 <= 50000):
    매수 = False
elif not (1.0 < 등락율 <= 20.0):
    매수 = False

# VI 및 라운드피겨 제외
elif not (현재가 < VI아래5호가):
    매수 = False
elif 라운드피겨위5호가이내:
    매수 = False

# 이동평균 정배열 조건 (핵심)
# 1) 단기 > 중기 > 장기 (5분 > 20분 > 60분)
elif not (이동평균(5) > 이동평균(20) > 이동평균(60)):
    매수 = False

# 2) 모든 이동평균선 상승 중
elif not (이동평균(5) > 이동평균(5, 1)):
    매수 = False
elif not (이동평균(20) > 이동평균(20, 1)):
    매수 = False
elif not (이동평균(60) > 이동평균(60, 1)):
    매수 = False

# 3) 현재가가 단기 이평 위에 위치
elif not (현재가 > 이동평균(5)):
    매수 = False

# 4) 단기와 중기 이평 간 적정 간격 유지 (급등 후 과열 제외)
elif not (이동평균(5) < 이동평균(20) * 1.05):
    매수 = False

# 분봉 패턴 조건
elif not (현재가 > 분봉시가):
    매수 = False
elif not (분봉고가 > 분봉고가N(1)):
    매수 = False

# 거래량 조건
elif not (분당거래대금 > 분당거래대금평균(30) * 1.2):
    매수 = False

# 체결강도 조건
elif not (체결강도 > 115):
    매수 = False
elif not (체결강도평균(20) > 110):
    매수 = False

# 등락율 각도 조건 (상승 추세)
elif not (등락율각도(20) > 5):
    매수 = False

# 시가총액별 세분화 조건
elif 시가총액 < 2500:
    # 소형주
    if not (분당매수수량 > 분당매도수량 * 1.5):
        매수 = False
    elif not (당일거래대금 > 50 * 100):
        매수 = False
    elif not (전일비 > 80):
        매수 = False
    elif not (전일동시간비 > 120):
        매수 = False
    elif not (회전율 > 2.5):
        매수 = False
    elif not (현재가 > 고가 - (고가 - 저가) * 20 / 100):
        매수 = False
    elif not ((현재가 / 저가 - 1) * 100 <= 10):
        매수 = False
    elif not (매수총잔량 > 매도총잔량 * 1.2):
        매수 = False

elif 시가총액 < 5000:
    # 중형주
    if not (분당매수수량 > 분당매도수량 * 1.3):
        매수 = False
    elif not (당일거래대금 > 100 * 100):
        매수 = False
    elif not (전일비 > 60):
        매수 = False
    elif not (전일동시간비 > 100):
        매수 = False
    elif not (회전율 > 2.0):
        매수 = False
    elif not (현재가 > 고가 - (고가 - 저가) * 25 / 100):
        매수 = False
    elif not ((현재가 / 저가 - 1) * 100 <= 8):
        매수 = False
    elif not (매수총잔량 > 매도총잔량):
        매수 = False

else:
    # 대형주
    if not (분당매수수량 > 분당매도수량 * 1.2):
        매수 = False
    elif not (당일거래대금 > 200 * 100):
        매수 = False
    elif not (전일비 > 50):
        매수 = False
    elif not (전일동시간비 > 80):
        매수 = False
    elif not (회전율 > 1.5):
        매수 = False
    elif not (현재가 > 고가 - (고가 - 저가) * 30 / 100):
        매수 = False
    elif not ((현재가 / 저가 - 1) * 100 <= 7):
        매수 = False
    elif not (매수총잔량 > 매도총잔량 * 0.9):
        매수 = False

# 최종 매수 실행
if 매수:
    self.Buy(종목코드, 종목명, 매수수량, 현재가, 매도호가1, 매수호가1, 데이터길이)
```

### 매도 조건식

```python
# 정배열 이탈 및 익절 전략
if 등락율 > 29.5:
    매도 = True
elif 수익률 >= 6.0:
    매도 = True
elif 수익률 <= -3.0:
    매도 = True
elif 최고수익률 - 수익률 >= 4.0:
    매도 = True
elif 보유시간 > 300:  # 300분 (5시간) 초과 보유 시 청산
    매도 = True

# 정배열 붕괴 조건 (핵심 매도 조건)
# 1) 단기 이평이 중기 이평 하회
elif 이동평균(5) < 이동평균(20):
    if 수익률 > 0:
        매도 = True

# 2) 단기 이평 하락 전환
elif 이동평균(5) < 이동평균(5, 1) and 이동평균(5, 1) < 이동평균(5, 2):
    if 수익률 > 1.0:
        매도 = True

# 3) 현재가가 단기 이평 하회
elif 현재가 < 이동평균(5):
    if 수익률 > 0.5:
        매도 = True

# 중기 이평 하락 전환 (추세 약화)
elif 이동평균(20) < 이동평균(20, 1):
    if 수익률 > 2.0:
        매도 = True

# 체결강도 약화
elif 체결강도 < 100:
    매도 = True
elif 체결강도평균(20) < 105:
    if 수익률 > 1.0:
        매도 = True

# 분봉 음봉 연속
elif 현재가 < 분봉시가 and 현재가N(1) < 분봉시가N(1):
    if 수익률 > 1.5:
        매도 = True

# 등락율 각도 하락
elif 등락율각도(20) < 0:
    if 수익률 > 0.5:
        매도 = True

# 시가총액별 세분화 매도 조건
elif 시가총액 < 2500:
    # 소형주: 적극적 익절
    if 3.0 <= 수익률 < 6.0 and 분당매도수량 > 분당매수수량 * 1.8:
        매도 = True
    elif 거래대금증감 < 0:
        매도 = True
    elif 분당거래대금 < 분당거래대금평균(30) * 0.7:
        매도 = True

elif 시가총액 < 5000:
    # 중형주
    if 3.0 <= 수익률 < 6.0 and 분당매도수량 > 분당매수수량 * 1.6:
        매도 = True
    elif 거래대금증감 < -50 * 100:
        매도 = True
    elif 분당거래대금 < 분당거래대금평균(30) * 0.75:
        매도 = True

else:
    # 대형주
    if 3.0 <= 수익률 < 6.0 and 분당매도수량 > 분당매수수량 * 1.5:
        매도 = True
    elif 거래대금증감 < -100 * 100:
        매도 = True
    elif 분당거래대금 < 분당거래대금평균(30) * 0.8:
        매도 = True

if 매도:
    self.Sell(종목코드, 종목명, 매도수량, 현재가, 매도호가1, 매수호가1, 강제청산)
```
